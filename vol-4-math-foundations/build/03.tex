\documentclass[tikz,border=8pt]{standalone}
\usepackage{amsmath,amssymb}
\usetikzlibrary{arrows.meta,calc,decorations.markings}

\begin{document}
\begin{tikzpicture}[>=Latex,
  divarrow/.style={->, red!70!black, very thick},
  curlarrow/.style={->, blue!70!black, very thick},
  flowarrow/.style={->, green!45!black, very thick}]

  % LEFT — divergence (radial outflow)
  \begin{scope}[xshift=0cm]
    \node[font=\bfseries] at (0, 2.4) {Divergence};
    \foreach \angle in {0,30,60,90,120,150,180,210,240,270,300,330}{
      \draw[divarrow] (\angle:0.45) -- (\angle:1.6);
    }
    \draw[thick, dashed] (-0.32,-0.32) rectangle (0.32,0.32);
    \node[font=\small] at (0.55, -0.45) {$V$};
    \node[align=center, font=\footnotesize] at (0,-2.5)
      {$\nabla\!\cdot\!\mathbf F=\lim\limits_{V\to0}\dfrac{1}{|V|}\!\oint_{\partial V}\!\mathbf F\!\cdot\!\mathrm d\mathbf A$};
  \end{scope}

  % MIDDLE — curl (tangential)
  \begin{scope}[xshift=5.5cm]
    \node[font=\bfseries] at (0, 2.4) {Curl};
    \foreach \angle in {0,30,60,90,120,150,180,210,240,270,300,330}{
      \pgfmathsetmacro{\sx}{1.3*cos(\angle)}
      \pgfmathsetmacro{\sy}{1.3*sin(\angle)}
      \pgfmathsetmacro{\dx}{0.55*cos(\angle+90)}
      \pgfmathsetmacro{\dy}{0.55*sin(\angle+90)}
      \draw[curlarrow] (\sx,\sy) -- ++ (\dx,\dy);
    }
    \draw[thick, dashed] (0,0) circle (0.45);
    \node[font=\small] at (0.62, -0.55) {$\partial S$};
    \node[align=center, font=\footnotesize] at (0,-2.5)
      {$\nabla\!\times\!\mathbf F=\lim\limits_{A\to0}\dfrac{1}{A}\!\oint_{\partial S}\!\mathbf F\!\cdot\!\mathrm d\mathbf l\,\hat{\mathbf n}$};
  \end{scope}

  % RIGHT — parallel flow
  \begin{scope}[xshift=11cm]
    \node[font=\bfseries] at (0, 2.4) {Parallel flow};
    \foreach \y in {-1.4,-0.7,0,0.7,1.4}{
      \foreach \x in {-1.4,-0.4,0.6}{
        \draw[flowarrow] (\x,\y) -- ++ (1.0,0);
      }
    }
    \node[align=center, font=\footnotesize] at (0,-2.5)
      {$\nabla\!\cdot\!\mathbf F=0,\ \nabla\!\times\!\mathbf F=\mathbf 0$};
  \end{scope}
\end{tikzpicture}
\end{document}
